## 알고리즘 1

~~~c++
#include <iostream>
using namespace std;

int bit[4] = {0,};

void print_array() {
    for(int i=0; i<4; i++)
        cout << bit[i] << " ";
    cout << endl;
}

int main() {
    int i,j,k,l;
    
    for(i=0; i<2; i++) {
        bit[0] = (i%2 == 0) ? 0:1;

        for(j=0; j<2; j++){
            bit[1] = (j%2 == 0) ? 0:1;

            for(k=0; k<2; k++) {
                bit[2] = (k%2 == 0) ? 0:1;

                for(l=0; l<2; l++) {
                    bit[3] = (l%2 == 0) ? 0:1;

                    print_array();
                }
            }
        }
    }
    return 0;
}
~~~

## 알고리즘 2

~~~c++
#include <iostream>
using namespace std;

int main() {

    int i, j;
    int arr[] = {1,2,4};
    int n = sizeof(arr)/sizeof(arr[0]); // n: 원소의 개수
    // 1<<n : 부분집합의 수
    int size = (1<<n);                  
    for(i=0; i < size; i++) { // 0~7

        // 원소의 수만큼 Bit를 비교
        for(j=0; j<n; j++) {            
            // i의 j번째 bit가 1이면 j번째 원소 출력 
            if( i & (1<<j) )         
                cout << arr[j] << " ";

        } cout << endl;
    }
    return 0;
}
~~~
